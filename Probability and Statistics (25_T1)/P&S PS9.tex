\section{Question 2b}
\begin{itemize}
    \item The joint distribution is summarized by the table below:
    \[
    \begin{array}{c|c|c|c}
         & Y=0 & Y=1 & Y=2 \\
        X=0 & 0.1 & 0.25 & 0.15 \\
        X=1 & 0.25 & 0.05 & 0.2
    \end{array}
    \]
    \item $p(X=0) = \sum_{i=0}^2 p(0,Y = i) = 0.1+0.25+0.15 = 0.5$
    \item $p(X=1) = 1-0.5 = 0.5$
    \item $p(Y=0) = \sum_{i=0}^1 p(X = i,0) = 0.1+0.25 =0.35$
    \item $p(Y=1) = 0.25+0.05 = 0.3$
    \item $p(Y=2) = 0.15+0.2 = 0.35$
\end{itemize}

\section{Question 3d}
\begin{itemize}
    \item We know $SER=\sqrt{\frac{SSR}{n-2}}, R^2 = 1 - \frac{SSR}{TSS}, TSS = \sum_{i=1}^n (Y_i - \overline{Y})^2$
    \item $SSR = (n-2)SER^2 = (100-2) \times 11.5^2 = 12961$
    \item $TSS = \frac{SSR}{1-R^2} = \frac{12961}{1-0.08} = 14088$
    \item The sample variance of test score $s^2 = \frac{TSS}{n-1} = \frac{14088}{99} = 142.3$ and the sample standard deviation is $s_{TestScore} = \sqrt{142.3} = 11.9$
\end{itemize}
